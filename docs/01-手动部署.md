## 手动部署sing-box
```bash
sudo apt update && sudo apt install curl tar
cd /tmp
curl -LO https://github.com/SagerNet/sing-box/releases/download/v1.13.11/sing-box-1.13.11-linux-amd64.tar.gz
tar -xzf sing-box-1.13.11-linux-amd64.tar.gz
sudo install -m 0755 sing-box-1.13.11-linux-amd64/sing-box /usr/local/bin/sing-box
sing-box version
sudo useradd --system --no-create-home --shell /usr/sbin/nologin sing-box

KEYPAIR=$(sing-box generate reality-keypair)
private_key=$(echo "$KEYPAIR" | awk 'NR==1 {print $2}')
public_key=$(echo "$KEYPAIR" | awk 'NR==2 {print $2}')
uuid=$(sing-box generate uuid)
short_id=$(sing-box generate rand 8 --hex)
sudo mkdir -p /etc/sing-box
sudo tee /etc/sing-box/config.json > /dev/null << EOF
{
  "log": {
    "level": "info"
  },
  "inbounds": [
    {
      "type": "vless",
      "tag": "vless-in",
      "listen": "0.0.0.0",
      "listen_port": 443,
      "users": [
        {
          "uuid": "$uuid",
          "flow": "xtls-rprx-vision"
        }
      ],
      "tls": {
        "enabled": true,
        "server_name": "www.microsoft.com",
        "reality": {
          "enabled": true,
          "handshake": {
            "server": "www.microsoft.com",
            "server_port": 443
          },
          "private_key": "$private_key",
          "short_id": [
            "$short_id"
          ]
        }
      }
    }
  ],
  "outbounds": [
    {
      "type": "direct",
      "tag": "direct"
    }
  ]
}
EOF

sudo chown root:sing-box /etc/sing-box
sudo chmod 0750 /etc/sing-box

sudo chown root:sing-box /etc/sing-box/config.json
sudo chmod 0640 /etc/sing-box/config.json

sudo tee /etc/systemd/system/sing-box.service > /dev/null << EOF
[Unit]
Description=sing-box service
Documentation=https://sing-box.sagernet.org/
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
User=sing-box
Group=sing-box

ExecStartPre=/usr/local/bin/sing-box check -c /etc/sing-box/config.json
ExecStart=/usr/local/bin/sing-box run -c /etc/sing-box/config.json

Restart=on-failure
RestartSec=5s

AmbientCapabilities=CAP_NET_BIND_SERVICE
CapabilityBoundingSet=CAP_NET_BIND_SERVICE
NoNewPrivileges=true

ProtectHome=true
PrivateTmp=true

LimitNOFILE=1048576

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable --now sing-box.service
sudo systemctl status sing-box.service
```
